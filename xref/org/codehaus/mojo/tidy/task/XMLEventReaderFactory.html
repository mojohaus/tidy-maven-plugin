22  import org.codehaus.stax2.XMLInputFactory2;
23  
24  import javax.xml.stream.XMLEventReader;
25  import javax.xml.stream.XMLInputFactory;
26  import javax.xml.stream.XMLStreamException;
27  import java.io.StringReader;
28  
29  class XMLEventReaderFactory
30  {
31      private static final XMLInputFactory XML_INPUT_FACTORY = createInputFactory();
32  
33      static XMLEventReader createEventReaderForPom( String pom )
34          throws XMLStreamException
35      {
36          return XML_INPUT_FACTORY.createXMLEventReader( new StringReader( pom ) );
37      }
38  
39      private static XMLInputFactory createInputFactory()
40      {
41          XMLInputFactory inputFactory = XMLInputFactory2.newInstance();
42          inputFactory.setProperty( XMLInputFactory2.P_PRESERVE_LOCATION, true );
43          return inputFactory;
44      }
45  }
