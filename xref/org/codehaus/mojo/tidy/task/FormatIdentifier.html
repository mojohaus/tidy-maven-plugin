22  import java.util.List;
23  
24  import static java.util.Arrays.asList;
25  
26  /**
27   * Identifies the output format for a given POM. It tries to find a
28   * format that is similar to the current format.
29   */
30  class FormatIdentifier
31  {
32      private static final List<String> LINE_SEPARATORS = asList( "\r\n", "\n", "\r" );
33  
34      /**
35       * Identifies the output format for the given POM.
36       *
37       * @param pom the POM.
38       * @return the output format.
39       */
40      Format identifyFormat( String pom )
41      {
42          for ( String separator : LINE_SEPARATORS )
43          {
44              if ( pom.contains( separator ) )
45              {
46                  return new Format( separator );
47              }
48          }
49  
50          throw new IllegalArgumentException( "The pom.xml has no known line separator." );
51      }
52  }
